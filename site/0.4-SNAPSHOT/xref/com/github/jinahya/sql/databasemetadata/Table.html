113     /**
114      *
115      * @param database
116      * @param suppression
117      * @param catalog
118      * @param schemaPattern
119      * @param tableNamePattern
120      * @param types
121      * @param tables
122      *
123      * @throws SQLException if a database access error occurs.
124      *
125      * @see DatabaseMetaData#getTables(String, String, String, String[])
126      */
127     public static void retrieve(final DatabaseMetaData database,
128                                 final Suppression suppression,
129                                 final String catalog,
130                                 final String schemaPattern,
131                                 final String tableNamePattern,
132                                 final String[] types,
133                                 final Collection<? super Table> tables)
134         throws SQLException {
135 
136         if (database == null) {
137             throw new NullPointerException("null database");
138         }
139 
140         if (suppression == null) {
141             throw new NullPointerException("null suppression");
142         }
143 
144         if (tables == null) {
145             throw new NullPointerException("null tables");
146         }
147 
148         if (suppression.isSuppressed(Schema.SUPPRESSION_PATH_TABLES)) {
149             return;
150         }
151 
152         final ResultSet resultSet = database.getTables(
153             catalog, schemaPattern, tableNamePattern, types);
154         try {
155             while (resultSet.next()) {
156                 tables.add(ColumnRetriever.retrieve(
157                     Table.class, suppression, resultSet));
158             }
159         } finally {
160             resultSet.close();
161         }
162     }

165     /**
166      *
167      * @param database
168      * @param suppression
169      * @param schema
170      *
171      * @throws SQLException if a database access error occurs.
172      */
173     public static void retrieve(final DatabaseMetaData database,
174                                 final Suppression suppression,
175                                 final Schema schema)
176         throws SQLException {
177 
178         if (database == null) {
179             throw new NullPointerException("null database");
180         }
181 
182         if (suppression == null) {
183             throw new NullPointerException("null suppression");
184         }
185 
186         if (schema == null) {
187             throw new NullPointerException("null schema");
188         }
189 
190         retrieve(database, suppression,
191                  schema.getCatalog().getTableCat(),
192                  schema.getTableSchem(),
193                  null,
194                  null,
195                  schema.getTables());
196 
197         for (final Table table : schema.getTables()) {
198             table.schema = schema;
199         }
200 
201         for (final Table table : schema.getTables()) {
202             BestRowIdentifier.retrieve(database, suppression, table);
203             Column.retrieve(database, suppression, table);
204             ColumnPrivilege.retrieve(database, suppression, table);
205             ExportedKey.retrieve(database, suppression, table);
206             ImportedKey.retrieve(database, suppression, table);
207             IndexInfo.retrieve(database, suppression, table);
208             PrimaryKey.retrieve(database, suppression, table);
209             TablePrivilege.retrieve(database, suppression, table);
210             VersionColumn.retrieve(database, suppression, table);
211         }
212     }

226

230

417     public List<String> getIndexColumnNames(final String indexName) {
418 
419         if (indexName == null) {
420             throw new NullPointerException("null indexName");
421         }
422 
423         final List<String> indexColumnNames = new ArrayList<String>();
424 
425         for (final IndexInfo indexInfo_ : getIndexInfo()) {
426             if (!indexName.equals(indexInfo_.getIndexName())) {
427                 continue;
428             }
429             indexColumnNames.add(indexInfo_.getColumnName());
430         }
431 
432         return indexColumnNames;
433     }

436     public List<String> getIndexNames(final Boolean nonUnique,
437                                       final Short... types) {
438 
439         final List<String> indexNames = new ArrayList<String>();
440 
441         for (final IndexInfo indexInfo_ : getIndexInfo()) {
442             if (nonUnique != null
443                 && indexInfo_.getNonUnique() != nonUnique.booleanValue()) {
444                 continue;
445             }
446             if (types != null) {
447                 boolean typeMatches = false;
448                 for (final Short type : types) {
449                     if (type == null) {
450                         throw new NullPointerException("null type");
451                     }
452                     if (type.shortValue() == indexInfo_.getType()) {
453                         typeMatches = true;
454                         break;
455                     }
456                 }
457                 if (!typeMatches) {
458                     continue;
459                 }
460             }
461             final String indexName = indexInfo_.getIndexName();
462             if (indexNames.contains(indexName)) {
463                 continue;
464             }
465             indexNames.add(indexName);
466         }
467 
468         return indexNames;
469     }
