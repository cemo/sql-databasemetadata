56      /**
57       *
58       * @param database
59       * @param suppression
60       * @param catalog
61       * @param schemaPattern
62       * @param tableNamePattern
63       * @param columnNamePattern
64       * @param columns
65       *
66       * @throws SQLException if a database access error occurs.
67       *
68       * @see DatabaseMetaData#getColumns(String, String, String, String)
69       */
70      public static void retrieve(final DatabaseMetaData database,
71                                  final Suppression suppression,
72                                  final String catalog,
73                                  final String schemaPattern,
74                                  final String tableNamePattern,
75                                  final String columnNamePattern,
76                                  final Collection<? super Column> columns)
77          throws SQLException {
78  
79          if (database == null) {
80              throw new NullPointerException("null database");
81          }
82  
83          if (suppression == null) {
84              throw new NullPointerException("null suppression");
85          }
86  
87          if (columns == null) {
88              throw new NullPointerException("null columns");
89          }
90  
91          if (suppression.isSuppressed(Table.SUPPRESSION_PATH_COLUMNS)) {
92              return;
93          }
94  
95          final ResultSet resultSet = database.getColumns(
96              catalog, schemaPattern, tableNamePattern, columnNamePattern);
97          try {
98              while (resultSet.next()) {
99                  columns.add(ColumnRetriever.retrieve(
100                     Column.class, suppression, resultSet));
101             }
102         } finally {
103             resultSet.close();
104         }
105     }

108     /**
109      *
110      * @param database
111      * @param suppression
112      * @param table
113      *
114      * @throws SQLException if a database access error occurs.
115      */
116     public static void retrieve(final DatabaseMetaData database,
117                                 final Suppression suppression,
118                                 final Table table)
119         throws SQLException {
120 
121         if (database == null) {
122             throw new NullPointerException("null database");
123         }
124 
125         if (suppression == null) {
126             throw new NullPointerException("null suppression");
127         }
128 
129         if (table == null) {
130             throw new NullPointerException("null table");
131         }
132 
133         retrieve(database, suppression,
134                  table.getSchema().getCatalog().getTableCat(),
135                  table.getSchema().getTableSchem(), table.getTableName(),
136                  null,
137                  table.getColumns());
138 
139         for (final Column column : table.getColumns()) {
140             column.table = table;
141         }
142     }
